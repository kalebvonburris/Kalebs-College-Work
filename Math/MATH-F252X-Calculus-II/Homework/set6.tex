    \paragraph*{76.} 
    $\int\sin^7(2x)\cos(2x)dx$
    \\
    \begin{mdframed}
        \begin{equation*}
            \begin{gathered}
                \int\sin^7(2x)\cos(2x)dx                        \\
                u = \sin(2x) \quad \frac{du}{2} = \cos(2x)dx    \\
                \Rightarrow \frac{1}{2} \int u^7du              
                = \frac{1}{2}\left(\frac{u^8}{8}\right) + C     
                = \boxed{\frac{\sin^8(2x)}{16} + C}
            \end{gathered}
        \end{equation*}
    \end{mdframed}

    \paragraph*{86.}
    $\int \sqrt{\sin x}\cos^3 x dx$
    \\
    \begin{mdframed}
        \begin{equation*}
            \begin{gathered}
                \int \sqrt{\sin x}\cos^3 x dx                 
                = \int \sqrt{\sin x} (1 - \sin^2 x) \cos x dx
            \end{gathered}
        \end{equation*}
    \end{mdframed}
